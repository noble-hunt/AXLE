CREATE TABLE public.achievements (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  user_id uuid NOT NULL,
  name text NOT NULL,
  description text NOT NULL,
  progress numeric NOT NULL DEFAULT 0,
  unlocked boolean NOT NULL DEFAULT false,
  updated_at timestamp with time zone DEFAULT now(),
  CONSTRAINT achievements_pkey PRIMARY KEY (id),
  CONSTRAINT achievements_user_id_fkey FOREIGN KEY (user_id) REFERENCES auth.users(id)
);
CREATE TABLE public.group_members (
  group_id uuid NOT NULL,
  user_id uuid NOT NULL,
  role text NOT NULL DEFAULT 'member'::text CHECK (role = ANY (ARRAY['member'::text, 'admin'::text])),
  joined_at timestamp with time zone DEFAULT now(),
  CONSTRAINT group_members_pkey PRIMARY KEY (group_id, user_id),
  CONSTRAINT group_members_group_id_fkey FOREIGN KEY (group_id) REFERENCES public.groups(id),
  CONSTRAINT group_members_user_id_fkey FOREIGN KEY (user_id) REFERENCES auth.users(id)
);
CREATE TABLE public.group_posts (
  id bigint NOT NULL DEFAULT nextval('group_posts_id_seq'::regclass),
  group_id uuid NOT NULL,
  author_id uuid NOT NULL,
  body text NOT NULL,
  meta jsonb,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT group_posts_pkey PRIMARY KEY (id),
  CONSTRAINT group_posts_group_id_fkey FOREIGN KEY (group_id) REFERENCES public.groups(id),
  CONSTRAINT group_posts_author_id_fkey FOREIGN KEY (author_id) REFERENCES auth.users(id)
);
CREATE TABLE public.group_reactions (
  post_id bigint NOT NULL,
  user_id uuid NOT NULL,
  emoji text NOT NULL,
  created_at timestamp with time zone DEFAULT now(),
  CONSTRAINT group_reactions_pkey PRIMARY KEY (post_id, user_id, emoji),
  CONSTRAINT group_reactions_post_id_fkey FOREIGN KEY (post_id) REFERENCES public.group_posts(id),
  CONSTRAINT group_reactions_user_id_fkey FOREIGN KEY (user_id) REFERENCES auth.users(id)
);
CREATE TABLE public.groups (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  owner_id uuid NOT NULL,
  name text NOT NULL,
  visibility text NOT NULL DEFAULT 'private'::text CHECK (visibility = ANY (ARRAY['private'::text, 'public'::text])),
  created_at timestamp with time zone DEFAULT now(),
  updated_at timestamp with time zone DEFAULT now(),
  description text,
  photo_url text,
  is_public boolean NOT NULL DEFAULT false,
  CONSTRAINT groups_pkey PRIMARY KEY (id),
  CONSTRAINT groups_owner_id_fkey FOREIGN KEY (owner_id) REFERENCES auth.users(id)
);
CREATE TABLE public.health_reports (
  id uuid NOT NULL DEFAULT gen_random_uuid(),

  date date NOT NULL,
  summary text,
  metrics jsonb NOT NULL DEFAULT '{}'::jsonb,
  suggestions ARRAY NOT NULL DEFAULT '{}'::text[],
  fatigue_score real,
  CONSTRAINT health_reports_pkey PRIMARY KEY (id),
  CONSTRAINT health_reports_user_id_fkey FOREIGN KEY (user_id) REFERENCES auth.users(id)
);
CREATE TABLE public.profiles (
  user_id uuid NOT NULL,
  username text,
  avatar_url text,
  created_at timestamp with time zone DEFAULT now(),
  first_name text,
  last_name text,
  date_of_birth date CHECK (date_of_birth IS NULL OR date_of_birth >= '1900-01-01'::date AND date_of_birth <= (CURRENT_DATE - '13 years'::interval)),
  updated_at timestamp with time zone DEFAULT now(),
  last_lat numeric,
  last_lon numeric,
  timezone text,
  preferred_unit text NOT NULL DEFAULT 'lbs'::text,
  favorite_movements ARRAY NOT NULL DEFAULT '{}'::text[],
  CONSTRAINT profiles_pkey PRIMARY KEY (user_id),
  CONSTRAINT profiles_user_id_fkey FOREIGN KEY (user_id) REFERENCES auth.users(id)
);

CREATE TABLE public.workout_feedback (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  workout_id uuid NOT NULL,
