Add a “More” card and modal:

// client/src/pages/WorkoutGenerate/FocusStep.tsx
// imports: add useState and your Modal/Sheet component
const PRIMARY = [
  { id:'strength',    title:'Strength',     desc:'Heavy compounds, lower reps, longer rest.' },
  { id:'conditioning',title:'Conditioning', desc:'Intervals, circuits, metabolic power.' },
  { id:'mixed',       title:'Mixed',        desc:'Blend of strength + conditioning.' },
  { id:'endurance',   title:'Endurance',    desc:'Longer efforts, aerobic capacity.' },
];

const MORE = [
  { id:'crossfit',              title:'CrossFit',                 desc:'Classic CF: EMOM/AMRAP/For Time.' },
  { id:'olympic_weightlifting', title:'Olympic Weightlifting',    desc:'Snatch & C&J, EMOM/E2:00 complexes.' },
  { id:'powerlifting',          title:'Powerlifting',             desc:'Squat/Bench/Deadlift strength focus.' },
  { id:'bb_full_body',          title:'Bodybuilding — Full Body', desc:'Hypertrophy across all patterns.' },
  { id:'bb_upper',              title:'Bodybuilding — Upper',     desc:'Chest/back/shoulders/arms volume.' },
  { id:'bb_lower',              title:'Bodybuilding — Lower',     desc:'Quads/hams/glutes volume.' },
  { id:'aerobic',               title:'Aerobic (Cardio)',         desc:'Z2–Z4 intervals or steady state.' },
  { id:'gymnastics',            title:'Gymnastics Work',          desc:'Skill EMOMs: TTB, MU, HS, strict pull.' },
  { id:'mobility',              title:'Mobility Session',         desc:'Quality mobility & tissue work.' }
];

export default function FocusStep() {
  const [showMore, setShowMore] = useState(false);
  const setFocus = useWizardStore(s => s.setFocus);

  return (
    <>
      <Section title="Choose your focus">
        <CardGrid>
          {PRIMARY.map(c => (
            <FocusCard key={c.id} title={c.title} desc={c.desc}
              onClick={()=> setFocus(c.id as Focus)} />
          ))}
          <FocusCard title="More" desc="CrossFit, Oly, PL, BB, Cardio, Gymnastics, Mobility"
            onClick={()=> setShowMore(true)} />
        </CardGrid>
      </Section>

      <Sheet open={showMore} onOpenChange={setShowMore} title="More categories">
        <CardGrid>
          {MORE.map(c => (
            <FocusCard key={c.id} title={c.title} desc={c.desc}
              onClick={()=> { setFocus(c.id as Focus); setShowMore(false); }} />
          ))}
        </CardGrid>
      </Sheet>
    </>
  );
}
